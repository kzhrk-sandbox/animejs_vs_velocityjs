anime.js vs velocity.js

open modal

anime.js(NG)

anime({
	targets: this.overlay,
	opacity: [0, 1],
	complete: ()=>{
		this.isOpen = true;
		this.isAnimate = false;
	}
});

velocity.js

velocity(this.overlay, {
	opacity: [1, 0],
},{
	complete: ()=>{
		this.isOpen = true;
		this.isAnimate = false;
	}
});

anime.js(GOOD)

anime({
	targets: this.overlay,
	opacity: [0, 1],
	easing: 'easeOutQuad',
	complete: ()=>{
		this.isOpen = true;
		this.isAnimate = false;
	}
});

アンカーリンクのスクロール

const scroll = {
	y: window.scrollY || window.pageYOffset
};

anime({
	targets: scroll,
	y: 0,
	duration: 350,
	easing: 'easeInOutCubic',
	update: () => {
		console.log(`scrollY: ${scroll.y}`);
		window.scrollTo(0, scroll.y);
	}
});
